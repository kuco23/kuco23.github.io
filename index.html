class SubClass(SuperClass):
    def __init__(self, *args, **kwargs):
        super(SubClass, self).__init__(args, kwargs)
	# putting SubClass, self as arguments in super is unecessary

# this works as inheritance, SubClass inherited the whole SuperClass.
# Now SubClass has every SuperClass function.

class SubClass(SuperClass1, Superclass2):
    def __init__(self, arg, *args, **kwargs):
	super(Subclass, self).__init__(arg, *args, **kwargs)

  Number Types:

  byte var /* must be an int between -128 and 127 [-2^7, 2^7-1] */ (8-bit value)
  short var /* int between -32 768 and 32 767 [-2^15, 2^15-1] (16 bit)

  byte, short save memory (byte is 4 times smaller than int, short 2 times)

  int var /* 32-bit signed integer between [-2 147 483 648, 2147 483 647] (2^32) */
  (int is considered a standard numeric declaration -> if you dont declare a var its considered int
  that is a problem when passing arguments between functions that have non int declarations)
  long var /* 64-bit signed [-2^63, 2^63-1] */
  float var /* single-precision 32-bit IEEE 754 floating point */
  double var /* double-precision 64-bit IEEE 754 floating point */

  default value = false for all number types

Bolean:
  boolean var /* one bit of information (true or false), default value=false */
  eg. boolean one = true

Char:
  char var /* single 16-bit Unicode character */
  eg. char letterA = 'A'

Reference Datatypes:
  default value of reference variables = null
